# 数组内对象去重

# reduce((累加器,当前值,当前索引,所调用的数组)=>{

# !!!必须要有返回值!!!

# },累加器第一次的初始值)

let arr = [
    {
        id:1,
        name:'小王'
    },
    {
        id:3,
        name:'小张'
    },
    {
        id:2,
        name:'小李'
    },
    {
        id:1,
        name:'小陈'
    },
    {
        id:1,
        name:'小红'
    }
]

let obj = {}
let data = arr.reduce((beforeValue,nowValue)=>{
    if(obj[nowValue.id]){
        return beforeValue //必须
    }else {
        obj[nowValue.id] = nowValue.name
        beforeValue.push(nowValue)
        return beforeValue //必须
    }
},[])
