# 盗用构造函数

# 可给父类传递参数

# 问题：父类的方法必须要在构造函数中定义,这意味着子类所继承的方法是每次调用时创建的，而不是指向同一个方法

function Parent(name){
    this.name = name
    this.sayName = function(){
        console.log(this.name)
    }
}

function Children(name,age){
    //继承属性
    Parent.call(this,name)
    this.age = age
}

let a = new Children("李四",20)
console.log("a",a)
a.sayName()//李四
let b = new Children("张三",20)
b.sayName()//张三

console.log(a.sayName==b.sayName)//false

# 组合继承

# 结合前两种方法的优点

可以给父类传参

保证子类继承的方法指向同一个

function Parent(name){
    this.name = name;
}
Parent.prototype.sayName = function(){
    console.log(this.name)
}

function Children(name,age){
    //继承属性
    Parent.call(this,name)
    this.age = age;
}
//继承方法
Children.prototype = new Parent()

let a = new Children("张三",19)
let b = new Children("李四",20)

console.log(a.sayName==b.sayName)//true

# 寄生式组合继承

# 这个 inherit()函数实现了寄生式组合继承的核心逻辑。这个函数接收两个参数：子 类构造函数和父类构造函数。在这个函数内部，第一步是创建父类原型的一个副本。然后，给返回的 prototype 对象设置 constructor 属性，解决由于重写原型导致默认 constructor 丢失的问题。最 后将新创建的对象赋值给子类型的原型。

function inherit(parent,children){
    let prototype = Object(parent.prototype)//创建对象
    prototype.constructor = children//增强对象
    children.prototype = prototype//赋值对象
}

function Parent(name){
    this.name = name
}
Parent.prototype.sayName = function(){
    console.log(this.name)
}

function Children(name,age){
    this.age = age
    Parent.call(this,name)
}

inherit(Parent,Children)

let a = new Children("小王",20)
console.log("a",a)
console.log("a instanceof Parent",a instanceof Parent)//true 继承于Parent
console.log("a instanceof Children",a instanceof Children)// true Children的实例
