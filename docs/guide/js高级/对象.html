# 对象

# 什么是对象

使用键对值的方式存储多条数据

# 为什么使用对象

一个变量名存储多条数据

# 取值的方式

方式一:对象.属性名

操作简单,有时不能用

let obj = {
    name:'小陈',
    age:22,
    setName:function(value){
        this.name = value
    }
}

obj.setName('小李')
console.log(obj.name)//小李

方式二:对象["属性名"]

操作复杂,通用

let obj = {
    name:"小陈",
    age:22,
	setName:function(value){
        this.name = value
    }
}

obj['setName']('小李')
console.log(obj['name'])//小李

# 什么时候使用第二种方式

1.当属性名使用特殊符号时,- 空格等

let obj = {};
obj.content-type = 'text/json';//报错
obj['content-type'] = 'text/json';
console.log(obj)//{content-type: 'text/json'}

2.当属性名不确定时

let obj = {};
let name = "myName";
let age = 22;
obj.name = age;
console.log(obj)//{name:22}
obj[name] = age;
console.log(obj)//{myName:22}
